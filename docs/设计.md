好的，我们以 `CreateEntityRagdoll` 作为**第一推动力**，从该钩子启动整个动画系统。以下是整合后的模块入口设计。

---

### 主入口 (`lua/autorun/server/init.lua`)

```lua
-- 加载所有模块
local Scheduler = include("modules/scheduler.lua")
local StateMachine = include("modules/state_machine.lua")
local AnimationSelector = include("modules/animation_selector.lua")

-- 全局 Tick 调度器启动
hook.Add("Tick", "EDAE_Scheduler", Scheduler.Tick)

-- 第一推动力：实体生成布娃娃时启动状态机
hook.Add("CreateEntityRagdoll", "EDAE_StartStateMachine", function(owner, physRag)
    -- 从 owner 提取伤害上下文
    local context = {
        hp         = owner:GetMaxHealth() or 100,
        damageType = owner.arDmg or "Bullet",
        hitGroup   = owner.arHit or 0,
        isHeadshot = owner.arHeadshot or false,
        hostiles   = owner.arHostiles or {},
        friends    = owner.arFriends or {},
    }

    -- 决定初始状态（通常为 DEATH）
    local initialState = StateMachine.State.DEATH

    -- 启动调度器，为该布娃娃创建一个协程运行状态机
    Scheduler.StartRagdoll(physRag, initialState, context)
end)
```

### 调度器 (`modules/scheduler.lua`)

**职责**：管理所有活跃布娃娃的协程，在 `Tick` 中恢复执行。

```lua
local Scheduler = {}
local coroutines = {}  -- [entIndex] = coroutine

function Scheduler.StartRagdoll(physRag, initialState, context)
    local index = physRag:EntIndex()
    if coroutines[index] then
        -- 若已有协程，先关闭旧的
        coroutine.close(coroutines[index])
    end

    local co = coroutine.create(function()
        StateMachine.Run(physRag, initialState, context)
        -- 状态机结束后清理
        coroutines[index] = nil
    end)

    coroutines[index] = co
end

function Scheduler.Tick()
    for index, co in pairs(coroutines) do
        if coroutine.status(co) == "dead" then
            coroutines[index] = nil
        else
            local success, err = coroutine.resume(co)
            if not success then
                print("Coroutine error: " .. tostring(err))
                coroutines[index] = nil
            end
        end
    end
end

return Scheduler
```

### 状态机 (`modules/state_machine.lua`)

**职责**：定义状态流转逻辑，每个状态内部调用 `AnimationPlayer.Play` 播放动画。

```lua
local AnimationPlayer = include("modules/animation_player.lua")
local AnimationSelector = include("modules/animation_selector.lua")

local StateMachine = {}

StateMachine.State = {
    DEATH  = 1,
    CRAWL  = 2,
    WRITHE = 3,
    REVIVE = 4,
    DONE   = 5,
}

local StateHandlers = {}

function StateHandlers[StateMachine.State.DEATH](physRag, ctx)
    -- 选择死亡动画
    local animName = AnimationSelector.Select(ctx)
    if not animName then
        return StateMachine.State.DONE, ctx
    end

    -- 播放动画（内部是协程，会逐帧 yield）
    local success = AnimationPlayer.Play(physRag, animName, nil, function()
        -- 每帧检测：血量归零则中断
        if ctx.hp <= 0 then
            return true  -- 返回 true 表示中断播放
        end
    end)

    -- 决定下一状态
    if not success or ctx.hp <= 0 then
        return StateMachine.State.DONE, ctx
    elseif ShouldCrawl(ctx) then
        return StateMachine.State.CRAWL, ctx
    else
        return StateMachine.State.DONE, ctx
    end
end

-- 其他状态类似...

function StateMachine.Run(physRag, initialState, ctx)
    local state = initialState
    while state and state ~= StateMachine.State.DONE do
        local handler = StateHandlers[state]
        if not handler then break end
        state, ctx = handler(physRag, ctx)
    end
end

return StateMachine
```

### 动画播放器 (`modules/animation_player.lua`)

**职责**：播放单次动画，返回成功或失败。

```lua
local EntityFactory = include("modules/entity_factory.lua")
local PhysicsDriver = include("modules/physics_driver.lua")

local AnimationPlayer = {}

function AnimationPlayer.Play(physRag, animName, duration, onFrame)
    local animRag = EntityFactory.CreateAnimEntity(physRag)
    if not animRag then return false end

    animRag:Fire("SetAnimation", animName, 0)
    local seqID = animRag:LookupSequence(animName)
    duration = duration or animRag:SequenceDuration(seqID) or 3.0

    PhysicsDriver.DisableMotion(physRag)
    coroutine.yield()
    PhysicsDriver.EnableMotion(physRag)

    local startTime = CurTime()
    local interrupted = false

    while true do
        if not IsValid(animRag) or not IsValid(physRag) then
            interrupted = true
            break
        end
        if CurTime() - startTime > duration then break end

        PhysicsDriver.ApplyBoneTransform(animRag, physRag)

        if onFrame then
            local shouldStop = onFrame()
            if shouldStop then
                interrupted = true
                break
            end
        end

        coroutine.yield()
    end

    if IsValid(animRag) then animRag:Remove() end
    return not interrupted
end

return AnimationPlayer
```

### 总结

- **第一推动力**：`CreateEntityRagdoll` 钩子触发，创建上下文并调用调度器。
- **调度器**：为每个布娃娃启动独立协程，运行状态机。
- **状态机**：决策动画选择与状态流转，调用动画播放器。
- **动画播放器**：纯协程，负责单次动画的逐帧驱动，支持外部中断。

整个流程线清晰，各模块职责单一，完全符合你的架构期望。
