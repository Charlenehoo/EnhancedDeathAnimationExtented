## 架构设计：骨骼级协程调度器

### 1. 核心概念

| 概念                | 说明                                                                                                           |
| ------------------- | -------------------------------------------------------------------------------------------------------------- |
| **PhysObj 协程**    | 每个物理骨骼对应一个独立协程，负责该骨骼的地形适应、障碍物检测和 CSC 驱动                                      |
| **Group（协程组）** | 属于同一个 Ragdoll 的所有 PhysObj 协程的集合，以 `physRag` 为键                                                |
| **调度器**          | 统一管理所有 Group，在 `Tick` 中按组遍历并恢复协程                                                             |
| **yield 返回值**    | 协程通过 `coroutine.yield(state)` 向调度器报告当前状态（如 `"fall"`、`"hitwall"`），调度器可据此做组级别的决策 |

### 3. 单个骨骼协程的模板

```lua
local function boneCoroutine(physRag, boneID, animRag)
    local physObj = physRag:GetPhysicsObjectNum(physRag:TranslateBoneToPhysBone(boneID))
    local boneName = physRag:GetBoneName(boneID)
    local animBoneID = animRag:LookupBone(boneName)

    -- 每骨骼独立的状态变量
    local addpos = Vector(0, 0, 0)
    local lastAdd = Vector(0, 0, 0)
    local lastHit = Vector(0, 0, 0)
    local isFall = false
    local isHitWall = false

    while true do
        -- 1. 获取动画目标位置
        local targetPos, targetAng = animRag:GetBonePosition(animBoneID)
        local refer = Vector(targetPos.x, targetPos.y, animRag:GetPos().z)

        -- 2. 射线检测地面
        local tr1 = util.TraceLine({
            start = refer + Vector(0,0,10),
            endpos = refer - Vector(0,0,100),
            mask = MASK_SOLID,
            filter = {physRag, animRag}
        })

        -- 3. 计算高度差和累积偏移
        local currentHit = refer - tr1.HitPos
        local diff = (refer.z - tr1.HitPos.z) - lastHit.z
        addpos = Vector(0, 0, diff + lastAdd.z)

        -- 4. 坠落检测
        if math.abs(diff) >= 20 then
            isFall = true
        end

        -- 5. 障碍物检测
        local bone_pos = targetPos - addpos
        local tr2 = util.TraceLine({
            start = physObj:GetPos(),
            endpos = bone_pos,
            mask = MASK_ALL,
            filter = {physRag, animRag}
        })
        isHitWall = tr2.Hit

        -- 6. 更新历史
        lastAdd = addpos
        lastHit = currentHit

        -- 7. 应用 CSC（仅在未坠落且未撞墙时）
        if not isFall and not isHitWall then
            local csc = {
                secondstoarrive = 0.01,
                pos = bone_pos,
                angle = targetAng,
                maxangular = 400,
                maxangulardamp = 200,
                maxspeed = 400,
                maxspeeddamp = 300,
                teleportdistance = 0
            }
            physObj:Wake()
            physObj:ComputeShadowControl(csc)
        end

        -- 8. 向调度器报告状态
        coroutine.yield({
            fall = isFall,
            hitWall = isHitWall
        })
    end
end
```

### 4. 调度器的 Tick 逻辑

```lua
local function tickScheduler()
    for physRag, group in pairs(groups) do
        if not IsValid(physRag) then
            groups[physRag] = nil
            continue
        end

        if not group.active then
            continue
        end

        local newFallCount = 0
        local newHitWallCount = 0

        for boneID, co in pairs(group.coroutines) do
            if coroutine.status(co) == "dead" then
                group.coroutines[boneID] = nil
                continue
            end

            local success, state = coroutine.resume(co)
            if success and state then
                if state.fall then
                    newFallCount = newFallCount + 1
                end
                if state.hitWall then
                    newHitWallCount = newHitWallCount + 1
                end
            end
        end

        group.fallCount = newFallCount
        group.hitWallCount = newHitWallCount

        -- 组级别的终止条件
        if newFallCount >= 5 or newHitWallCount >= table.Count(group.coroutines) then
            group.active = false
            -- 可触发状态机转换到 DONE
        end
    end
end
```

### 5. 组的创建与销毁

```lua
local function createGroup(physRag, animRag)
    local group = {
        coroutines = {},
        fallCount = 0,
        hitWallCount = 0,
        active = true
    }

    for i = 0, physRag:GetPhysicsObjectCount() - 1 do
        local boneID = physRag:TranslatePhysBoneToBone(i)
        if boneID then
            local co = coroutine.create(function()
                boneCoroutine(physRag, boneID, animRag)
            end)
            group.coroutines[boneID] = co
        end
    end

    groups[physRag] = group
    return group
end

local function killGroup(physRag)
    groups[physRag] = nil
    -- 协程会被 GC 自动清理
end
```

### 6. 设计优势

| 原始代码                                      | 新架构                                  |
| --------------------------------------------- | --------------------------------------- |
| 所有骨骼的逻辑挤在一个大循环中                | 每个骨骼独立协程，逻辑线性清晰          |
| 状态变量散落在 `ORag.arMoveBone[boneName]` 中 | 状态是协程的局部变量，天然隔离          |
| 坠崖/撞墙计数通过修改全局变量实现             | 调度器通过 `yield` 返回值收集状态并汇总 |
| 难以单独暂停某个骨骼                          | 可通过不恢复特定协程实现                |
| 整个 Ragdoll 的 Tick 逻辑无法复用             | 每个骨骼协程可独立测试和替换            |

### 7. 与状态机的集成

```lua
-- 在状态机的死亡动画状态中
function StateHandlers[State.DEATH](physRag, ctx)
    local animRag = createAnimRag(physRag)
    local group = createGroup(physRag, animRag)

    -- 等待动画结束或组被标记为 inactive
    while group.active and CurTime() - startTime < duration do
        coroutine.yield()
    end

    killGroup(physRag)
    -- 决定下一状态
end
```

这个设计将原始代码中**隐式的、纠缠的状态机**显式化为**协程 + 调度器**的结构，既保留了每骨骼独立的精细控制，又通过分组实现了统一的生命周期管理。
